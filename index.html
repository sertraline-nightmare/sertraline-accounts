const GUILD_ID = 'YOUR_GUILD_ID';
const BOT_TOKEN = 'YOUR_BOT_TOKEN';

const res = await fetch('https://raw.githubusercontent.com/sertraline-nightmare/sertraline-accounts/main/discord_accounts.json');
const data = await res.json();
const ids = data.userIds.split(' ');

for (const id of ids) {
  await fetch(`https://discord.com/api/v10/guilds/${GUILD_ID}/bans/${id}`, {
    method: 'PUT',
    headers: {
      'Authorization': `Bot ${BOT_TOKEN}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      delete_message_seconds: 604800,
      reason: 'سيرترالين وأعوانه — sertraline-accounts',
    }),
  });
}
